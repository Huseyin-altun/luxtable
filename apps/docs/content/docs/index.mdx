## Column Filtering

Enable column-level filtering with `options.filtering`. Use dropdown filters for columns with limited values!

<LuxTableFilterDemo />

```tsx
import { LuxTable, createColumnHelper } from "luxtable";

const columnHelper = createColumnHelper<Employee>();

const columns = [
  // Text filter (default)
  columnHelper.accessor("name", { header: "Name" }),
  
  // Dropdown filter for limited value columns
  columnHelper.accessor("department", { 
    header: "Department",
    meta: { filterVariant: "select" }, // Dropdown filter!
  }),
  
  columnHelper.accessor("status", { 
    header: "Status",
    meta: { filterVariant: "select" }, // Dropdown filter!
  }),
];

export default function App() {
  return (
    <LuxTable 
      data={data} 
      columns={columns} 
      options={{ filtering: true }}
    />
  );
}
```

## Click to Copy Feature

Use `CopyableCell` to make any cell value copyable with a single click!

<LuxTableCopyableDemo />

```tsx
import { LuxTable, createColumnHelper, CopyableCell } from "luxtable";

const columnHelper = createColumnHelper<User>();

const columns = [
  columnHelper.accessor("email", {
    header: "Email",
    cell: (info) => <CopyableCell value={info.getValue()} />,
  }),
];
```

## Simple Example

```tsx
import { LuxTable, createColumnHelper, StatusCell } from "luxtable";

const data = [
  { name: "John", status: "Active" },
  { name: "Jane", status: "Pending" },
];

const columnHelper = createColumnHelper();

const columns = [
  // No cell needed - auto renders!
  columnHelper.accessor("name", { header: "Name" }),
  // Use built-in StatusCell for beautiful badges
  columnHelper.accessor("status", {
    header: "Status",
    cell: (info) => <StatusCell value={info.getValue()} />,
  }),
];

export default function App() {
  return <LuxTable data={data} columns={columns} />;
}
```
