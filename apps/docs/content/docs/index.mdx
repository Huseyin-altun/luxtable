## Simple Example

```tsx
import { LuxTable, createColumnHelper, StatusCell } from "luxtable";

const data = [
  { name: "John", status: "Active" },
  { name: "Jane", status: "Pending" },
];

const columnHelper = createColumnHelper();

const columns = [
  // No cell needed - auto renders!
  columnHelper.accessor("name", { header: "Name" }),
  // Use built-in StatusCell for beautiful badges
  columnHelper.accessor("status", {
    header: "Status",
    cell: (info) => <StatusCell value={info.getValue()} />,
  }),
];

export default function App() {
  return <LuxTable data={data} columns={columns} />;
}
```
