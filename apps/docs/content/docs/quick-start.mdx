# Quick Start

Get up and running with LuxTable in just a few minutes.

## Installation

```bash
npm install luxtable
# or
yarn add luxtable
# or
pnpm add luxtable
```

## Method 1: createColumnHelper (Recommended)

The most flexible method. Define columns one by one. **If you don't specify a cell, the value is automatically rendered**:

<LuxTableDemo />

```tsx
"use client";

import { LuxTable, createColumnHelper, StatusCell, ProgressCell } from "luxtable";

type Person = {
  firstName: string;
  lastName: string;
  age: number;
  status: string;
  progress: number;
};

const data: Person[] = [
  { firstName: "Tanner", lastName: "Linsley", age: 24, status: "Active", progress: 50 },
  { firstName: "Jane", lastName: "Doe", age: 32, status: "Pending", progress: 80 },
];

const columnHelper = createColumnHelper<Person>();

const columns = [
  // ðŸŽ‰ No cell definition needed - auto render!
  columnHelper.accessor("firstName", { header: "First Name" }),
  columnHelper.accessor("lastName", { header: "Last Name" }),
  columnHelper.accessor("age", { header: "Age" }),
  // Use cell for columns where you want custom appearance
  columnHelper.accessor("status", {
    header: "Status",
    cell: (info) => <StatusCell value={info.getValue()} />,
  }),
  columnHelper.accessor("progress", {
    header: "Progress",
    cell: (info) => <ProgressCell value={info.getValue()} showLabel />,
  }),
];

export function MyTable() {
  return <LuxTable data={data} columns={columns} options={{ pagination: true }} />;
}
```

## Method 2: createColumnsFromData - Auto from JSON

**Fastest method!** Provide your JSON data, columns are generated automatically. Only specify where you want custom rendering:

<LuxTableAutoDemo />

```tsx
"use client";

import { LuxTable, createColumnsFromData, StatusCell, ProgressCell } from "luxtable";

const data = [
  { firstName: "Tanner", lastName: "Linsley", age: 24, status: "Active", progress: 50 },
  { firstName: "Jane", lastName: "Doe", age: 32, status: "Pending", progress: 80 },
];

// ðŸš€ Columns ready in one line!
const columns = createColumnsFromData(data, {
  // Custom headers (optional - camelCase is automatically converted to Title Case)
  headers: {
    firstName: "First Name",
    lastName: "Last Name",
  },
  // Only specify columns requiring custom appearance
  cells: {
    status: (info) => <StatusCell value={info.getValue() as string} />,
    progress: (info) => <ProgressCell value={info.getValue() as number} showLabel />,
  },
  // Hide unwanted columns (optional)
  // exclude: ["id", "createdAt"],
});

export function MyTable() {
  return <LuxTable data={data} columns={columns} />;
}
```
